cmake_minimum_required(VERSION 3.13)

# Define el nombre de tu proyecto
project(Lavadora)

# Agrega el ejecutable principal
add_executable(Lavadora 
    main.c 
    centrifugado.c
)

# Enlaza las bibliotecas necesarias (si las hay)
target_link_libraries(Lavadora pico_stdlib)

# Habilita la salida USB y deshabilita la salida UART
pico_enable_stdio_usb(Lavadora 1)
pico_enable_stdio_uart(Lavadora 0)

# Agrega salidas adicionales si es necesario
pico_add_extra_outputs(Lavadora)
